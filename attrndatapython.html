def check_missing_data(attrn_data):
    flag=attrn_data.isna().sum().any()
    if flag==True:
        total = attrn_data.isnull().sum()
        percent = (attrn_data.isnull().sum())/(attrn_data.isnull().count()*100)
        output = pd.concat([total, percent], axis=1, keys=['Total', 'Percent'])
        data_type = []
        for col in attrn_data.columns:
            dtype = str(attrn_data[col].dtype)
            data_type.append(dtype)
        output['Types'] = data_type
        return(np.transpose(output))
    else:
        return(False)
